  <script>
    const profileDropdown = document.getElementById('profileDropdown');
    const dropdownContent = document.getElementById('dropdownContent');

    profileDropdown.addEventListener('click', (e) => {
      e.stopPropagation();
      dropdownContent.style.display = dropdownContent.style.display === 'block' ? 'none' : 'block';
    });

    document.addEventListener('click', () => {
      dropdownContent.style.display = 'none';
    });

    // Fetch GitHub user info
    document.addEventListener("DOMContentLoaded", async () => {
      try {
        const res = await fetch('/api/me');
        if (res.ok) {
          const data = await res.json();

          // Update username
          document.getElementById("githubUsername").textContent = `Hi, ${data.username}`;

          // Update main profile icon
          document.getElementById("profileIcon").src = data.avatarUrl;

          // Update dropdown profile icon
          document.getElementById("dropdownProfileIcon").src = data.avatarUrl;
        }
      } catch (err) {
        console.error("Error fetching user info:", err);
      }
    });
  </script>
